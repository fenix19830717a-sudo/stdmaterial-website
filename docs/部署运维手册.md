## 2. 服务器初始化配置

### 2.1 创建非 root 用户

```bash
# 创建新用户
sudo adduser deploy

# 添加到 sudo 组
sudo usermod -aG sudo deploy

# 切换到 deploy 用户
su - deploy
```

### 2.2 配置 SSH 密钥

```bash
# 在本地生成 SSH 密钥
ssh-keygen -t ed25519 -C "your_email@example.com"

# 将公钥复制到服务器
ssh-copy-id deploy@your_server_ip

# 测试登录
ssh deploy@your_server_ip
```

### 2.3 禁用密码登录和 root 登录

```bash
sudo nano /etc/ssh/sshd_config
```

修改以下配置：

```
PermitRootLogin no
PasswordAuthentication no
PubkeyAuthentication yes
```

然后重启 SSH 服务：

```bash
sudo systemctl restart sshd
```

### 2.4 配置防火墙

```bash
# 安装 UFW
sudo apt install ufw

# 配置规则
sudo ufw allow 22    # SSH
sudo ufw allow 80    # HTTP
sudo ufw allow 443   # HTTPS

# 启用防火墙
sudo ufw enable
```

### 2.5 安装 Docker

```bash
# 更新包索引
sudo apt update
sudo apt install -y ca-certificates curl gnupg lsb-release

# 添加 Docker GPG 密钥
sudo mkdir -p /etc/apt/keyrings
curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /etc/apt/keyrings/docker.gpg

# 添加 Docker 仓库
echo "deb [arch=$(dpkg --print-architecture) signed-by=/etc/apt/keyrings/docker.gpg] https://download.docker.com/linux/ubuntu $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null

# 安装 Docker
sudo apt update
sudo apt install -y docker-ce docker-ce-cli containerd.io docker-compose-plugin

# 启动并启用 Docker
sudo systemctl start docker
sudo systemctl enable docker

# 将 deploy 用户添加到 docker 组
sudo usermod -aG docker deploy
```

## 4. Nginx 配置详解

### 4.1 主配置文件

`nginx.conf`:

```nginx
user nginx;
worker_processes auto;
error_log /var/log/nginx/error.log warn;
pid /var/run/nginx.pid;

events {
    worker_connections 1024;
    use epoll;
    multi_accept on;
}

http {
    include /etc/nginx/mime.types;
    default_type application/octet-stream;

    # 日志格式
    log_format main '$remote_addr - $remote_user [$time_local] "$request" '
                    '$status $body_bytes_sent "$http_referer" '
                    '"$http_user_agent" "$http_x_forwarded_for"';

    access_log /var/log/nginx/access.log main;

    # 性能优化
    sendfile on;
    tcp_nopush on;
    tcp_nodelay on;
    keepalive_timeout 65;
    types_hash_max_size 2048;

    # Gzip 压缩
    gzip on;
    gzip_vary on;
    gzip_proxied any;
    gzip_comp_level 6;
    gzip_types text/plain text/css text/xml application/json application/javascript application/xml+rss application/atom+xml image/svg+xml;

    # 静态资源缓存
    open_file_cache max=10000 inactive=30s;
    open_file_cache_valid 60s;
    open_file_cache_min_uses 2;
    open_file_cache_errors on;

    # 上游服务
    upstream openclaw {
        server openclaw:8080;
    }

    # 隐藏 Nginx 版本
    server_tokens off;

    include /etc/nginx/conf.d/*.conf;
}
```

### 4.2 站点配置文件

在 `conf.d/` 目录下创建 `shengtongda.conf`:

```nginx
# HTTP 重定向到 HTTPS
server {
    listen 80;
    server_name www.shengtongda.com shengtongda.com;
    
    return 301 https://$server_name$request_uri;
}

# HTTPS 配置
server {
    listen 443 ssl http2;
    server_name www.shengtongda.com shengtongda.com;

    # SSL 证书配置
    ssl_certificate /etc/nginx/ssl/fullchain.pem;
    ssl_certificate_key /etc/nginx/ssl/privkey.pem;
    
    # SSL 安全配置
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256:ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384;
    ssl_prefer_server_ciphers off;
    ssl_session_cache shared:SSL:10m;
    ssl_session_timeout 1d;
    ssl_session_tickets off;

    # HSTS
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;

    # 根目录
    root /usr/share/nginx/html;
    index index.html;

    # 访问日志
    access_log /var/log/nginx/shengtongda_access.log;

    # 安全头
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;

    # 静态资源缓存
    location ~* \.(jpg|jpeg|png|gif|ico|css|js|svg|woff|woff2|ttf|eot)$ {
        expires 30d;
        add_header Cache-Control "public, immutable";
    }

    # HTML 不缓存
    location ~* \.html$ {
        expires -1;
        add_header Cache-Control "no-store, no-cache, must-revalidate";
    }

    # OpenClaw API 代理
    location /api/openclaw/ {
        proxy_pass http://openclaw;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
        
        # 超时设置
        proxy_connect_timeout 60s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;
    }

    # AI 聊天功能
    location /chat/ {
        proxy_pass http://openclaw;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_cache_bypass $http_upgrade;
    }

    # 默认路由 - 尝试文件，否则返回 index.html
    location / {
        try_files $uri $uri/ /index.html;
    }

    # 禁止访问隐藏文件
    location ~ /\. {
        deny all;
        access_log off;
        log_not_found off;
    }
}
```

## 6. 备份与恢复策略

### 6.1 备份脚本

创建 `backup.sh`:

```bash
#!/bin/bash

# 配置
BACKUP_DIR="/home/deploy/backups"
DATE=$(date +%Y%m%d_%H%M%S)
SITE_DIR="/home/deploy/shengtongda"
RETENTION_DAYS=30

# 创建备份目录
mkdir -p $BACKUP_DIR

# 备份网站文件
echo "开始备份网站文件..."
tar -czf $BACKUP_DIR/site_$DATE.tar.gz -C $SITE_DIR .

# 备份 Docker 卷（如有）
echo "备份 Docker 数据卷..."
docker run --rm -v shengtongda-data:/data -v $BACKUP_DIR:/backup alpine tar -czf /backup/data_$DATE.tar.gz /data

# 备份环境变量（排除敏感信息）
echo "备份配置..."
cp $SITE_DIR/.env $BACKUP_DIR/.env.$DATE

# 备份 Nginx 配置
cp /etc/nginx/conf.d/shengtongda.conf $BACKUP_DIR/nginx.conf.$DATE

# 清理旧备份
echo "清理过期备份..."
find $BACKUP_DIR -name "*.tar.gz" -mtime +$RETENTION_DAYS -delete
find $BACKUP_DIR -name ".env.*" -mtime +$RETENTION_DAYS -delete

echo "备份完成: $DATE"
```

## 7. 监控与告警

### 7.1 基本监控脚本

创建 `monitor.sh`:

```bash
#!/bin/bash

# 告警配置
ALERT_EMAIL="admin@shengtongda.com"
TEAM_WEBHOOK="https://oapi.dingtalk.com/robot/send?access_token=YOUR_TOKEN"

# 检查 Docker 容器状态
check_docker() {
    CONTAINERS=$(docker-compose ps -q)
    for CONTAINER in $CONTAINERS; do
        STATUS=$(docker inspect --format='{{.State.Status}}' $CONTAINER)
        if [ "$STATUS" != "running" ]; then
            send_alert "容器 $CONTAINER 状态异常: $STATUS"
        fi
    done
}

# 检查磁盘空间
check_disk() {
    USAGE=$(df -h / | awk 'NR==2 {print $5}' | sed 's/%//')
    if [ $USAGE -gt 80 ]; then
        send_alert "磁盘使用率: $USAGE%"
    fi
}

# 检查内存
check_memory() {
    MEMORY=$(free | grep Mem | awk '{printf "%.0f", $3/$2 * 100}')
    if [ $MEMORY -gt 85 ]; then
        send_alert "内存使用率: $MEMORY%"
    fi
}

# 检查网站可访问性
check_website() {
    HTTP_CODE=$(curl -s -o /dev/null -w "%{http_code}" https://www.shengtongda.com)
    if [ "$HTTP_CODE" != "200" ]; then
        send_alert "网站返回状态码: $HTTP_CODE"
    fi
}

# 发送告警
send_alert() {
    MESSAGE="[晟通达监控告警] $(date '+%Y-%m-%d %H:%M:%S') - $1"
    echo "$MESSAGE"
    
    # 发送邮件
    echo "$MESSAGE" | mail -s "晟通达网站告警" $ALERT_EMAIL
    
    # 发送钉钉通知（可选）
    # curl -X POST $TEAM_WEBHOOK -H 'Content-Type: application/json' -d "{\"msgtype\":\"text\",\"text\":{\"content\":\"$MESSAGE\"}}"
}

# 执行检查
check_docker
check_disk
check_memory
check_website
```
